\begin{algorithm}[t]
\caption{Terminal Velocity Matching (TVM) Training}
\label{alg:terminal_velocity}
\begin{algorithmic}[1]
\Require Data $x$, noise $\epsilon$, displacement network $F_\theta$, velocity network $u_\theta$, optimizer
\State Sample $t, s \sim \text{Uniform}(0,1)$
\State $x_t \gets (1 - t) \, x + t \, \epsilon$; \quad $x_s \gets (1 - s) \, x + s \, \epsilon$
\State $v_s \gets \epsilon - x$ \Comment{Conditional velocity at $s$}
\State $f_{t,s} \gets (s - t) \, F_\theta(x_t, t, s)$ \Comment{Displacement map}
\State Compute JVP: $\frac{d}{ds} F_\theta(x_t, t, s)$
\State $\text{term\_vel} \gets F_\theta(x_t, t, s) + (s - t) \cdot \frac{d}{ds} F_\theta$ \Comment{Terminal velocity}
\State $x_{\text{arrived}} \gets x_t + f_{t,s}$ \textbf{detach}
\State $u_{\text{tgt}} \gets u_\theta(x_{\text{arrived}}, s)$ \textbf{detach}
\State $\mathcal{L}_{\text{TV}} \gets \| \text{term\_vel} - u_{\text{tgt}} \|^2$ \Comment{Terminal velocity loss}
\State $u_{\text{pred}} \gets u_\theta(x_s, s)$
\State $\mathcal{L}_{\text{FM}} \gets \| u_{\text{pred}} - v_s \|^2$ \Comment{Flow matching loss}
\State $\mathcal{L} \gets \mathcal{L}_{\text{TV}} + \mathcal{L}_{\text{FM}}$
\State Update $\theta$ with $\nabla_\theta \mathcal{L}$
\State \Return $\mathcal{L}$
\end{algorithmic}
\end{algorithm}
