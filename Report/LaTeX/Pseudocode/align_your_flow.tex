\begin{algorithm}[t]
\caption{Align Your Flow (AYF) Distillation}
\label{alg:align_your_flow}
\begin{algorithmic}[1]
\Require Data $x$, noise $\epsilon$, teacher velocity $v_{\text{teacher}}$, flow map $f_\theta$, optimizer
\State Sample $s, t \sim \text{Uniform}(0,1)$ with $s < t$
\State $z_t \gets (1 - t) \, \epsilon + t \, x$
\State $\eta \gets \delta \cdot (s - t)$ \Comment{Small step size}
\State $v_t \gets v_{\text{teacher}}(z_t, t)$
\State $z_{t'} \gets z_t + \eta \cdot v_t$; \quad $t' \gets t + \eta$ \Comment{Teacher Euler step}
\State $f_{s,t} \gets f_\theta(z_t, t, s)$
\State $f_{s,t'} \gets f_\theta(z_{t'}, t', s)$
\State $\mathcal{L}_{\text{EMD}} \gets \| f_{s,t} - f_{s,t'} \|^2$ \Comment{Eulerian Map Distillation}
\State Compute $\frac{\partial}{\partial t} f_\theta(z_t, t, s)$ via JVP
\State $v_{\text{avg}} \gets (f_{s,t} - z_t) / (s - t)$
\State $\mathcal{L}_{\text{LMD}} \gets \| \partial_t f_\theta - v_{\text{avg}} \|^2$ \Comment{Lagrangian Map Distillation}
\State $\mathcal{L} \gets \mathcal{L}_{\text{EMD}} + \mathcal{L}_{\text{LMD}}$
\State Update $\theta$ with $\nabla_\theta \mathcal{L}$
\State \Return $\mathcal{L}$
\end{algorithmic}
\end{algorithm}
