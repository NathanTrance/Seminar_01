\begin{algorithm}[htbp]
\caption{Consistency Flow Matching Training}
\label{alg:consistency_fm}
\begin{algorithmic}[1]
\Require $x \sim p_{\text{data}}$, $\epsilon \sim \mathcal{N}(0,I)$, velocity network $v_\theta$, EMA model $v_{\theta^-}$, optimizer, hyperparameter $\lambda$
\State Sample $t \sim \mathcal{U}[0, 1 - \Delta t]$
\State $z_t \gets (1 - t) \, x + t \, \epsilon$
\State $z_{t+\Delta t} \gets (1 - (t+\Delta t)) \, x + (t+\Delta t) \, \epsilon$
\State $v_t \gets v_\theta(z_t, t)$
\State $v_{t+\Delta t} \gets v_\theta(z_{t+\Delta t}, t+\Delta t)$
\State $f_t \gets z_t + (1 - t) \, v_t$
\State $f_{t+\Delta t} \gets z_{t+\Delta t} + (1 - (t+\Delta t)) \, v_{t+\Delta t}$
\State \Comment{with no grad}
\State $f_{t+\Delta t}^{\text{ema}} \gets z_{t+\Delta t} + (1 - (t+\Delta t)) \, v_{\theta^-}(z_{t+\Delta t}, t+\Delta t)$
\State $f_t^{\text{ema}} \gets z_t + (1 - t) \, v_{\theta^-}(z_t, t)$
\State $\mathcal{L}_f \gets \| f_t - f_{t+\Delta t}^{\text{ema}} \|^2 + \| f_{t+\Delta t} - f_t^{\text{ema}} \|^2$
\State $\mathcal{L}_v \gets \| v_t - v_{\theta^-}(z_{t+\Delta t}, t+\Delta t) \|^2 + \| v_{t+\Delta t} - v_{\theta^-}(z_t, t) \|^2$
\State $\mathcal{L} \gets \mathcal{L}_f + \lambda \, \mathcal{L}_v$
\State Update $\theta$ with $\nabla_\theta \mathcal{L}$; update EMA model
\State \Return $\mathcal{L}$
\end{algorithmic}
\end{algorithm}
