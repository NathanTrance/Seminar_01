\begin{algorithm}[htbp]
\caption{Meta Flow Map (MFM) Training}
\label{alg:meta_flow_maps}
\begin{algorithmic}[1]
\Require Data $x \sim p_{\text{data}}$, noise $\epsilon \sim \mathcal{N}(0,I)$, meta flow map $X_{\text{mfm}}$, optimizer
\State Sample $s, w, u \sim \mathcal{U}[0,1]$ with $s < w < u$
\State Build stochastic interpolants $I_s, I_w, I_u$ with shared noise $\epsilon$
\State $v_{u,u} \gets X_{\text{mfm}}(I_u, u, u)$ \Comment{Instantaneous velocity (diagonal)}
\State $\dot{I}_u \gets \dot{\alpha}_u \, x + \dot{\beta}_u \, \epsilon + \dot{\gamma}_u \, \xi$ \Comment{$\xi$ = extra randomness}
\State $\mathcal{L}_{\text{diag}} \gets \| v_{u,u} - \dot{I}_u \|^2$ \Comment{Diagonal flow matching loss}
\State $I_w \gets X_{\text{mfm}}(I_s, s, w)$ \Comment{Map $s \to w$}
\State $I_{u}^{\text{composed}} \gets X_{\text{mfm}}(I_w, w, u)$ \Comment{Map $w \to u$}
\State $I_{u}^{\text{direct}} \gets X_{\text{mfm}}(I_s, s, u)$ \Comment{Direct map $s \to u$}
\State $\mathcal{L}_{\text{cons}} \gets \| I_{u}^{\text{composed}} - I_{u}^{\text{direct}} \|^2$ \Comment{Consistency (semigroup) loss}
\State $\mathcal{L} \gets \mathcal{L}_{\text{diag}} + \mathcal{L}_{\text{cons}}$
\State Update parameters with $\nabla \mathcal{L}$
\State \Return $\mathcal{L}$
\end{algorithmic}
\end{algorithm}
